Code Example

-- Your first Nova game
function update(dt)
    if nova_key_down("right") then
        player.x = player.x + 200 * dt
    end
end

function draw()
    nova_2d_clear(10, 10, 40)
    nova_2d_rect(player.x, player.y, 50, 50, 255, 100, 50)
end
